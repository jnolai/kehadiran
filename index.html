Konfigurasi Google Sheets

Masukkan URL Google Apps Script Web App yang anda jana daripada Google Sheet anda.

Panduan Kod Google Apps Script
  1. Buka Google Sheet baru.
  2. Klik Extensions > Apps Script dan tampal kod di bawah.
  3. Deploy sebagai Web App (Akses set kepada: Anyone), salin URL dan tampal di kotak input atas.
function doPost(e) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    var data = JSON.parse(e.postData.contents);
    sheet.appendRow([
      new Date(), 
      data.email, 
      data.nama, 
      data.jawatan, 
      data.jabatan,
      data.catatan
    ]);
    return ContentService.createTextOutput(JSON.stringify({"status":"success"}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({"status":"error", "message": err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
